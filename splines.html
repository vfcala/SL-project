require(splines)
## Loading required package: splines
set.seed(123)
n = 100 
x = sort(runif(n))
m.x = sin(8*x - 4) + 2*exp(-(16^2)*(x - 0.5)^2)
y = rnorm(n, m.x, 0.3)
plot(x=x, y=y, main='data generating process')
lines(x, m.x)

plot(x=x, y=y, main='polynomial regressions')
# degree 0
lines(x, fitted(lm(y ~ 1)), col='green')
# degree 1
lines(x, fitted(lm(y ~ x)), col='red')
# degree 2
lines(x, fitted(lm(y ~ poly(x,2))), col='lightblue')
# degree 3
lines(x, fitted(lm(y ~ poly(x, degree=3))), col='darkgreen')
# degree 15
lines(x, fitted(lm(y ~ poly(x, degree=15))), col='purple')

###  (STEP FUNCTION)
##   divido il supporto di X [0,1] in 10 intervalli equispaziati
knots = seq(0.1, 0.9, by=0.1)
xcut = cut(x, c(-Inf, knots, Inf))
plot(x=x, y=y, main='piecewise polynomial regression')
abline(v=knots, lty=2)
##   in ogni intervallo calcolo il valor medio di y 
for (i in 1:length(levels(xcut))) {
  sub <- (xcut==levels(xcut)[i])
  lines(x[sub], fitted( lm( y[sub] ~ 1) ) )
}
#### piecewise cubic regression 
for (i in 1:length(levels(xcut))) {
  sub <- (xcut==levels(xcut)[i])
  lines(x[sub], fitted( lm( y[sub] ~ poly(x[sub], degree=3)) ), col='red' )
}

#### cubic spline & natural cubic spline with omonymous package 
plot(x=x, y=y, main='cubic splines: basic and natural')
lines(x, fitted(lm(y ~ bs(x,knots=knots))), col='green')
lines(x, fitted(lm(y ~ ns(x,knots=knots))), col='red')

#### SMOOTHING SPLINES in penalized minimization problem 
###  risolve il problema di QUANTI knots e DOVE posizionarli
##   lambda = 0 <-> zero-complexity-penalty; il fit Ã¨ un'interpolazione
##   lambda = Inf <-> curvatura impossibile; il fit Ã¨ una retta
plot(x=x, y=y, main='smoothing splines - penalized min problem')
fit_0 <- smooth.spline(x, y, lambda=0)
lines(x, predict(fit_0)$y, col=1)           # interpolazione
fit_inf <- smooth.spline(x, y, lambda=10^2)
lines(x, predict(fit_inf)$y, col=2)         # retta
